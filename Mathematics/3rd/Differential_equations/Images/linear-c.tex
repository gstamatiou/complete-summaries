\documentclass{standalone}
\usepackage{../../../../preamble_formulas}

\begin{document}
\begin{tikzpicture}[decorated curves/.style={
        postaction={
            decorate,
            decoration={
                markings,
                mark=at position 0.333 with {\arrow[black,scale=1.5]{stealth}},
                mark=at position 0.666 with {\arrow[black,scale=1.5]{stealth}}}}
      },decorated curves rev/.style={
        postaction={
            decorate,
            decoration={
                markings,
                mark=at position 0.333 with {\arrow[black,scale=1.5]{stealth[reversed]}},
                mark=at position 0.666 with {\arrow[black,scale=1.5]{stealth[reversed]}}}}
      },decorated separatrices rev/.style={
        postaction={
            decorate,
            decoration={
                markings,
                mark=at position 0.55 with {\arrow[scale=1.5]{stealth[reversed]}},
              }}
      },decorated separatrices/.style={
        postaction={
            decorate,
            decoration={
                markings,
                mark=at position 0.5 with {\arrow[scale=1.5]{stealth}},
              }}
      }]
  \begin{axis}[
      axis lines=middle,
      xmin=-4, xmax=4,
      ymin=-4, ymax=4,
      xtick=\empty, ytick=\empty,
      ylabel={$y$},
      xlabel={$x$},
      %axis on top,
      hide axis,
    ]
    \def\xmax{\pgfkeysvalueof{/pgfplots/xmax}}
    \def\xmin{\pgfkeysvalueof{/pgfplots/xmin}}
    \def\ymax{\pgfkeysvalueof{/pgfplots/ymax}}
    \def\ymin{\pgfkeysvalueof{/pgfplots/ymin}}
    \draw[black, thick,decorated separatrices rev] (0,0) --  (\xmin,0);
    \draw[black, thick,decorated separatrices rev] (0,0) --  (\xmax,0);
    \draw[black, thick,decorated separatrices] (0,\ymin) --(0,0);
    \draw[black, thick,decorated separatrices]  (0,\ymax) -- (0,0);
    \addplot[thick,samples=100,black,domain=0:\xmax,decorated separatrices rev]{x/3};
    \addplot[thick,samples=100,black,domain=\xmin:0,decorated separatrices]{x/3};
    \addplot[thick,samples=100,black,domain=\ymin/2:0,decorated separatrices]{2*x};
    \addplot[thick,samples=100,black,domain=0:\ymax/2,decorated separatrices rev]{2*x};
    \addplot[thick,samples=100,black,domain=0:-\ymin,decorated separatrices rev]{-x};
    \addplot[thick,samples=100,black,domain=-\ymax:0,decorated separatrices]{-x};
    \addplot[thick,samples=100,black,domain=-\ymax/3:0,decorated separatrices]{-3*x};
    \addplot[thick,samples=100,black,domain=0:-\ymin/3,decorated separatrices rev]{-3*x};
    \def\N{60}
    \def\ell{(\xmax-\xmin)/\N}
    \addplot[thin,mark=*,only marks, mark size=\ell/2*0.852*2cm,black] coordinates {
        (0,0)
      };
  \end{axis}
\end{tikzpicture}

\end{document}
