\begin{document}

\begin{tikzpicture}
    \begin{axis}[xmin=0,xmax=5,ymin=0,ymax=5,
    axis equal image]
    \addplot [thin,samples=200,smooth,blue] {f(x)};
    \addplot [thin,samples=200,smooth,black] {x};
    %%%%% Red lines 1
    \pgfplotsextra{
    \def\x{1.65}
    \def\y{0}
    \draw [very thin,red] (axis cs:\x,\y) \foreach \i in {0,...,10}{
    \pgfextra{
        \pgfkeys{/pgf/fpu=true, /pgf/fpu/output format=fixed}
        \pgfmathparse{f(\x)}\xdef\y{\pgfmathresult}
    }
    -- (axis cs:\x,\y) 
    \pgfextra{\xdef\x{\y}}

    };}
